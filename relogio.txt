function relogio(){
var data=new Date();
var h = data.getHours();
var min = data.getMinutes();
var seg = data.getSeconds();

if(h < 10){
  h ="0" + h;
 }
  if(min < 10){
   min = "0" +min;
  }
  if(seg <10){
  seg = "0" +seg;
  }
  var horas = h +":"+min+":"+ seg;
  document.querySelector(".tempo").value=horas;
}

var tempo=setInterval(relogio,1000);
